## 快速幂

```cpp
using ll = long long;
ll qpow(ll a, ll b, ll p) {
	ll res = 1 % p; // p == 1
	a %= p;

	for (; b > 0; b >>= 1) {
		if (b & 1)
			res = res * a % p;

		a = a * a % p;
	}
	return res;
}
```


## 二分

开区间 `(l, r)` 写法，重点维护循环不变量。

查找第一个满足条件的位置

```cpp
// F F F T T T
//     l r

int binary_search_min(vector<int> &nums, int x) {
	auto check = [&] (int i) {
		return nums[i] >= x;
	};

	int l = -1, r = nums.size();
	// check(l) == false
	// check(r) == true

	while (l + 1 < r) {
		int mid = l + (r - l) / 2;
		if (check(mid))
			r = mid;
		else
			l = mid;
	}
	return r;
}
```

查找最后一个满足条件的位置

```cpp
// T T T F F F
//     l r

int binary_search_max(vector<int> &nums, int x) {
	auto check = [&] (int i) {
		return nums[i] <= x;
	};

	int l = -1, r = nums.size();
	// check(l) == true
	// check(r) == false

	while (l + 1 < r) {
		int mid = l + (r - l) / 2;
		if (check(mid))
			l = mid;
		else
			r = mid;
	}
	return l;
}
```

## 差分

差分与前缀和“互为逆运算”：差分数组的前缀和还原原数组。

```cpp
vector<int> nums;
int n = nums.size();
vector<long long> diff(n + 1);

// 给左闭右开区间 [l, r) 都加上 v
auto add = [&] (int l, int r, int v) {
	diff[l] += v;
	diff[r] -= v;
};

// 前缀和重建原数组
auto build = [&] () {
	for (int i = 1; i < n; i++)
		diff[i] += diff[i - 1];

};
```

## ST 表

ST 表解决 **静态** 区间查询，包括区间最大 / 最小 / gcd
要求操作满足可重复/覆盖不影响答案。

`O(nlogn)` 预处理，`O(1)` 查询

定义 `f[k][i]` 表示从 i 开始，长度为 `2^k` 的子数组中的最大值
可以得到转移方程
`f[k][i] = max(f[k - 1][i], f[k - 1][i + (1 << (k - 1))])`
从长度为 `2 ^ (k - 1)` 的两段子数组计算出来

```cpp
using ll = long long;

struct ST {
	int K;
	vector<vector<ll>> st;

	ST(vector<int> &a) {
		int n = a.size();
		K = __lg(n) + 1;
		st.assign(K, vector<ll>(n));

		for (int i = 0; i < n; i++) {
			st[0][i] = a[i];
		}

		for (int k = 1; k < K; k++) {
			for (int i = 0; i + (1 << k) <= n; i++) {
				st[k][i] = max(st[k - 1][i],
				               st[k - 1][i + (1 << (k - 1))]);
			}
		}
	}

	// 闭区间 [l, r]
	ll query(int l, int r) {
		int k = __lg(r - l + 1);
		return max(st[k][l], st[k][r - (1 << k) + 1]);
	}
};
```


## 字典树

```cpp
struct Trie {
	struct Node {
		int son[26]{};
		bool end = false;
	};

	vector<Node> tr(1);

	void insert(const string & s) {
		int cur = 0;
		for (char c : s) {
			int d = c - 'a';
			if (tr[cur].son[d] == 0) {
				tr[cur].son[d] = tr.size();
				tr.emplace_back();
			}
			cur = tr[cur].son[d];
		}
		tr[cur].end = true;
	}

	bool find(const string & s) {
		int cur = 0;
		for (char c : s) {
			int d = c - 'a';
			if (tr[cur].son[d] == 0)
				return false;
			cur = tr[cur].son[d];
		}
		return tr[cur].end;
	}
};
```

## 树状数组

```c++
template<typename T>
class BIT {
	vector<T> tree;

public:
	// 使用下表 1 到 n
	BIT(int n) : tree(n + 1) {}

	// a[i] 增加 val
	// 1 <= i <= n
	void inc(int i, T val) {
		for (; i < tree.size(); i += i & -i)
			tree[i] += val;
	}

	// 求前缀和 a[1] + ... + a[i]
	T pre(int i) const {
		T res{};
		for (; i > 0; i -= i & -i)
			res += tree[i];
		return res;
	}

	// 求区间和 a[l] + ... + a[r]
	// 1 <= l <= r <= n
	T query(int l, int r) const {
		return pre(r) - pre(l - 1);
	}
};
```



# 图论

## 最短路

### Dijkstra

适用：非负边权的单源最短路。复杂度 `O((n + m) log n)`。

```cpp
// {x, weight}
vector<vector<pair<int, int>>> g;
int n = g.size();

auto dijkstra = [&] (int start) -> vector<long long> {
	vector<long long> dist(n, LLONG_MAX / 2);
	// {dist[x], x}
	priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<>> pq;

	dist[start] = 0;
	pq.emplace(0, start);
	while (!pq.empty()) {
		auto [d, x] = pq.top();
		pq.pop();

		if (d != dist[x]) continue;

		for (auto [y, w] : g[x]) {
			if (dist[y] > d + w) {
				dist[y] = d + w;
				pq.emplace(dist[y], y);
			}
		}
	}
	return dist;
};

```

### Floyd

适用：多源最短路，节点数较小。复杂度 `O(n^3)` (n <= 300)
`dist[i][j]` 为 i 到 j 最短距离
Floyd 的 k 是“当前新增允许的中转点”，所以必须最外层枚举。

```cpp
int n;
vector<vector<int>> edges; // u, v, w

vector dist(n, vector<long long>(n, LLONG_MAX / 2));

void floyd() {
	// dist[i][i] = 0
	for (int i = 0; i < n; i++) {
		dist[i][i] = 0;
	}

	// 初始化边
	for (auto & e : edges) {
		int u = e[0], v = e[1], w = e[2];
		dist[u][v] = w;
		dist[v][u] = w;
	}

	// 遍历 k -> update i, j
	for (int k = 0; k < n; k++) {
	    for (int i = 0; i < n; i++) {
	        for (int j = 0; j < n; j++) {
	            dist[i][j] = min(dist[i][j], dist[i][k] + dist[k][j]);
	        }
	    }
	}
}
```


## LCA

树上最近公共祖先，常用倍增做法。

核心定义：

- `pa[j][x]`：节点 `x` 往上跳 `2^j` 步后的祖先。
- `depth[x]`：节点 `x` 到根的边数。
- `pa[j][x] = pa[j - 1][pa[j - 1][x]]`，表示跳两段 `2^(j-1)`。

```cpp
class TreeAncestor {
    vector<int> depth;
    vector<vector<int>> pa;

public:
	TreeAncestor(vector<pair<int, int>> &edges) {
		int n = edges.size() + 1;
		int m = bit_width((unsigned) n);
		vector<vector<int>> g(n);
		for (auto &[x, y] : edges) {
			g[x].push_back(y);
			g[y].push_back(x);
		}

		depth.resize(n);
		pa.resize(m, vector<int>(n, -1));

		auto dfs = [&] (this auto && self, int x, int fa) -> void {
			pa[0][x] = fa;
			for (int y : g[x]) {
				if (y != fa) {
					depth[y] = depth[x] + 1;
					self(y, x);
				}
			}
		};
		dfs(0, -1);

		for (int j = 1; j < m; j++) {
			for (int x = 0; x < n; x++) {
				int p = pa[j - 1][x];
				if (p != -1)
					pa[j][x] = pa[j - 1][p];
			}
		}
	}

	int ancestor(int x, int k) {
		for (int j = 0; j < pa.size() && x != -1; j++) {
			if (k >> j & 1)
				x = pa[j][x];
		}

		return x;
	}

    // 返回 x, y的最近公共祖先
    int lca(int x, int y) {
		if (depth[x] < depth[y])
			swap(x, y);

		x = ancestor(x, depth[x] - depth[y]);
        if (x == y)
	        return x;

        for (int j = pa.size() - 1; j >= 0; j--) {
	        int px = pa[j][x], py = pa[j][y];
	        if (px != py) {
			    x = px;
			    y = py;
			}
        }
        return pa[0][x];
    }

    // 返回 x, y 之间的距离
    int dist(int x, int y) {
        int z = lca(x, y);
        return depth[x] + depth[y] - 2 * depth[z];
    }
};
```

# 数学
## 素数

### 朴素判断

适用：只判断少量数字是否为素数。单个数复杂度 `O(sqrt(x))`。

```cpp
bool isPrime(int x) {
    if (x < 2) return false;

    for (int i = 2; i * i <= x; i++) {
        if (x % i == 0) {
            return false;
        }
    }

    return true;
}
```

### 埃氏筛

适用：预处理 `[1, n]` 中哪些数是质数。

```cpp
const int MX = 1'000'010;
vector<bool> is_prime(MX, true);

auto init = []() {
    is_prime[0] = is_prime[1] = false;
    for (int i = 2; i * i < MX; i++) {
        if (is_prime[i]) {
            for (int j = i * i; j < MX; j += i) {
                is_prime[j] = false;
            }
        }
    }
    return 0;
}();
```

### 欧拉筛

欧拉筛也叫线性筛。核心性质：每个合数只会被它的最小质因子筛掉一次。

适用：需要预处理素数列表，或者想把筛法复杂度稳定到 `O(n)`。

```cpp
const int MX = 1'000'010;
vector<int> primes;
vector<bool> is_prime(MX, true);

auto init = []() {
    is_prime[0] = is_prime[1] = false;

    for (int i = 2; i < MX; i++) {
        if (is_prime[i]) {
            primes.push_back(i);
        }

        for (int p : primes) {
            if ((long long)i * p >= MX) {
                break;
            }

            is_prime[i * p] = false;

            // p 是 i 的最小质因子时，i * p 已经被最小质因子筛掉。
            // 继续用更大的质数筛，会破坏“每个合数只筛一次”。
            if (i % p == 0) {
                break;
            }
        }
    }
    return 0;
}();
```

## 质因数分解

### 朴素分解

适用：只分解少量数字，或者数据范围不大时。单个数复杂度 `O(sqrt(x))`。

只保留不同质因数：

```cpp
vector<int> primeFactors(int x) {
    vector<int> res;

    for (int i = 2; i * i <= x; i++) {
        if (x % i == 0) {
            res.push_back(i);
            while (x % i == 0) {
                x /= i;
            }
        }
    }

    if (x > 1) {
        res.push_back(x);
    }

    return res;
}
```

说明：循环结束后如果 `x > 1`，剩下的 `x` 一定是质数。

### SPF

`SPF = Smallest Prime Factor`，最小质因子。
如果 `x >= 2 && spf[x] == x`，说明 `x` 是质数。

适用：需要反复分解很多数字的质因数。
- SPF 预处理：`O(M log log M) 左右
- 单次分解：`O(log x)` 接近 O(质因数个数)

```cpp
const int MX = 1'000'010;
vector<int> spf(MX);

auto init = []() {
    for (int i = 0; i < MX; i++) {
        spf[i] = i;
    }

    for (int i = 2; i * i < MX; i++) {
        if (spf[i] == i) {
            for (int j = i * i; j < MX; j += i) {
                if (spf[j] == j) {
                    spf[j] = i;
                }
            }
        }
    }
    return 0;
}();

bool isPrime(int x) {
    return x >= 2 && spf[x] == x;
}

vector<int> primeFactors(int x) {
    vector<int> res;
    while (x > 1) {
        int p = spf[x];
        res.push_back(p);

        while (x % p == 0) {
            x /= p;
        }
    }
    return res;
}
```

# 动态规划 DP

## LIS
找到最长严格递增的子序列的长度
`O(n^2)`
**状态转移方程**：`dp[i] = max(dp[i], dp[j] + 1)`，其中 j < i 且 `nums[j] < nums[i]`

```cpp
int lengthOfLIS(vector<int> &nums) {
	int n = nums.size();
	vector<int> dp(n, 1);

	for (int i = 0; i < n; i++) {
		for (int j = 0; j < i; j++) {
			if (nums[j] < nums[i])
				dp[i] = max(dp[i], dp[j] + 1);
		}
	}

	return ranges::max(dp);
}
```

`O(nlog n)`
维护一个 `g` 数组，`g[i]` 存储所有长度为 `i+1` 的递增子序列中最小的尾部元素
严格递增时，相等不能扩展长度，所以要用 lower_bound 把相等位置替换掉。
最长不递减子序列，可以用 upper_bound

```cpp
int lis(vector<int>& nums) {
    int n = nums.size();
    vector<int> dp(n);
    vector<int> g;

    for (int i = 0; i < n; i++) {
        int x = nums[i];
        auto it = ranges::lower_bound(g, x);

        dp[i] = it - g.begin() + 1; // 以 i 结尾的 LIS 长度

        if (it == g.end())
            g.push_back(x);
        else
            *it = x;
    }
    return g.size();
}
```

如果只需要 LIS 总长度，答案是 `ranges::max(dp)`，也等于扫描结束后的 `g.size()`。

## 数位 DP

适用：统计 `[1, n]` 中满足某种数位性质的数字个数。

```cpp
class Solution {
public:
    int rotatedDigits(int m) {
        string s = to_string(m);
        int n = s.size();

        vector dp(n, vector(2, -1));
        auto dfs = [&] (this auto &&self, int i, bool has_diff, bool is_limit, bool is_num) -> int {
            if (i == n)
                return is_num && has_diff;

            if (!is_limit && is_num && dp[i][has_diff] != -1) {
                return dp[i][has_diff];
            }

            int ans = 0;
            if (!is_num) // 跳过这位不填数字
                ans += self(i + 1, false, false, false);

            int up = is_limit ? s[i] - '0' : 9; // 是limit, 这位最高填s[i] - '0', 否则填9
            int low = is_num ? 0 : 1;           // 是is_num, 这位填0, 否则只能填1(不然全是前导0了)

            // 枚举第i位填d
            for (int d = low; d <= up; d++) {
                if (d == 3 || d == 4 || d == 7) continue;
                bool ndiff = has_diff || d == 2 || d == 5 || d == 6 || d == 9;
                ans += self(i + 1, ndiff, is_limit && d == up, true);
            }

            if (!is_limit && is_num)
                dp[i][has_diff] = ans;

            return ans;
        };

        // 初始化 is_limit = true, 受到约束, 第一位最多填s[0]
        // 初始化 is_num = false, 受到约束, 第一位不能填0
        return dfs(0, false, true, false);
    }
};
```

参数：
- `i`：当前填到第几位
- `state`：前面数位留下的信息
- `isLimit`：是否贴着上界
- `isNum`：是否已经开始填数字

例子：
- `804`：`state = hasDiff`，不能含 `3/4/7`，至少含一个 `2/5/6/9`

# 字符串

## 字符串哈希
本质上把字符串看成base进制的数字
s = "abc", 哈希值为
`a * base ^ 2 + b * base + c`

```cpp
// h[i + 1] = h[i] * base + s[i];

using ull = unsigned long long;

struct StringHash {
	static const ull base = 131;
	vector<ull> h, p;

	StringHash(const string &s) {
		int n = s.size();
		h.resize(n + 1, 0);
		p.resize(n + 1, 1);

		for (int i = 0; i < n; i++) {
			h[i + 1] = h[i] * base + s[i];
			p[i + 1] = p[i] * base;
		}
	}

	// 返回闭区间 [l, r] 的哈希
	ull get(int l, int r) {
		return h[r + 1] - h[l] * p[r - l + 1];
	}
};

```

双哈希版本，冲突概率更低。

```cpp
using ll = long long;

struct StringHash {
	static const ll mod1 = 1000000007;
	static const ll mod2 = 1000000009;
	static const ll base = 131;

	vector<ll> h1, h2, p1, p2;

	StringHash(const string &s) {
		int n = s.size();
		h1.assign(n + 1, 0);
		h2.assign(n + 1, 0);
		p1.assign(n + 1, 1);
		p2.assign(n + 1, 1);

		for (int i = 0; i < n; i++) {
			int x = s[i];
			h1[i + 1] = (h1[i] * base + x) % mod1;
			h2[i + 1] = (h2[i] * base + x) % mod2;
			p1[i + 1] = p1[i] * base % mod1;
			p2[i + 1] = p2[i] * base % mod2;
		}
	}

	// 返回闭区间 [l, r] 的哈希
	pair<ll, ll> get(int l, int r) {
		int len = r - l + 1;
		ll x1 = (h1[r + 1] - h1[l] * p1[len] % mod1 + mod1) % mod1;
		ll x2 = (h2[r + 1] - h2[l] * p2[len] % mod2 + mod2) % mod2;
		return {x1, x2};
	}
};
```

## Z 函数
定义 `z[i] = LCP(s[i:], s)`，也就是从 `i` 开始的后缀和原串的最长公共前缀长度。

常用于字符串匹配：构造 `pattern + "#" + text`，如果某个位置的 `z[i] >= pattern.size()`，说明这里匹配到了模式串。

```cpp
vector<int> z_algorithm(const string &s) {
	int n = s.size();
	vector<int> z(n);
	z[0] = n;
	int l = 0, r = 0; // 当前已知最靠右的 z-box，范围是 [l, r)

	for (int i = 1; i < n; i++) {
		// 如果 i 落在 z-box 里面，可以先复用对称位置 i - l 的答案。
		// 但不能超过右边界 r，所以要和 r - i 取 min。
		if (i < r) {
			z[i] = min(r - i, z[i - l]);
		}

		// 从当前已知长度继续暴力扩展，比较 s[0...] 和 s[i...]。
		while (i + z[i] < n && s[z[i]] == s[i + z[i]]) {
			z[i]++;
		}

		// 如果以 i 开头的匹配段更靠右，就更新 z-box。
		if (i + z[i] > r) {
			l = i;
			r = i + z[i];
		}
	}

	return z;
}
```

例子：`s = "abacaba"`，`z = [7, 0, 1, 0, 3, 0, 1]`。
