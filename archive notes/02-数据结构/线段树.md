```cpp
struct InfoSum {
    long long x = 0;
};
InfoSum operator+(const InfoSum &a, const InfoSum &b) {
    return InfoSum{a.x + b.x};
}

struct InfoMax {
    long long x = LLONG_MIN;
};
InfoMax operator+(const InfoMax &a, const InfoMax &b) {
    return InfoMax{max(a.x b.x)};
}

template<class Info>
class SegmentTree{
    int n;
    vector<Info> info;

    void init(const vector<Info> &_init){
        n = (int)_init.size();
        info.assign((n * 4) + 1, Info());

        /// l, r is the range managed by current p
        /// p is current node index in tree
        function<void(int, int, int)> build = [&](int p, int l, int r){
            if (l == r){
                info[p] = _init[l - 1];
                return;
            }
            int mid = (l + r) / 2;
            build(2 * p, l, mid);
            build(2 * p + 1, mid + 1, r);
            pull(p);
        };
        build(1, 1, n);
    }

    void pull(int p){
        info[p] = info[2 * p] + info[2 * p + 1];
    }

    /// Single point update a[x] = v;
    /// p is current node index
    /// l, r is the range managed by current node index
    void modify(int p, int l, int r, int x, const Info &v){
        if (l == r){
            info[p] = v;
            return;
        }
        int mid = (l + r) / 2;
        if (x <= mid)
            modify(2 * p, l, mid, x, v);
        else
            modify(2 * p + 1, mid + 1, r, x, v);
        pull(p);
    }

    Info query(int p, int l, int r, int x, int y){
        if (l > y || r < x)
            return Info();

        if (l >= x && r <= y)
            return info[p];

        int mid = (l + r) / 2;
        return query(2 * p, l, mid, x, y) + query(2 * p + 1, mid + 1, r, x, y);
    }

public:
    SegmentTree() {}

    SegmentTree(int n, Info _init = Info()){
        init(vector<Info>(n, _init));
    }

    SegmentTree(const vector<Info> &_init){
        init(_init);
    }

    /// a[x] = v;
    void modify(int x, const Info &v){
        modify(1, 1, n, x, v);
    }

    /// range query [l, r]
    Info query(int l, int r){
        return query(1, 1, n, l, r);
    }

    /// a[x]
    Info at(int p) {
        assert(p >= 1 && p < n);
        return query(1, 1, n, p, p);
    }
};
```
