# 字典树

https://leetcode.com/problems/sum-of-prefix-scores-of-strings

```cpp
struct Trie {
    struct TrieNode {
        TrieNode *son[26]{};
        bool isEndWord = false;
    };

    TrieNode *root;

    Trie() : root(new TrieNode()) {
    }

    void insert(const string &s) {
        TrieNode *curt = root;
        for (char c : s) {
            c -= 'a';
            if (!curt->son[c])
                curt->son[c] = new TrieNode();
            curt = curt->son[c];
        }
        curt->isEndWord = true;
    }

    bool search(const string &s) {
        TrieNode *curt = root;
        for (char c : s) {
            c -= 'a';
            if (!curt->son[c])
                return false;
            curt = curt->son[c];
        }
        return curt->isEndWord;
    }
};
```
