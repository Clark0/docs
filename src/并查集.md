# DSU

```cpp
vector<int> pa;
vector<int> size;

void init(int n) {
    pa.resize(n);
    iota(pa.begin(), pa.end(), 0);
    size.resize(n, 1);
}

int find(int x) {
    return pa[x] == x ? x : pa[x] = find(pa[x]);
}

void unite(int x, int y) {
    x = find(x), y = find(y);
    pa[x] = y;
}

void unite2(int x, int y) {
    x = find(x), y = find(y);
    if (x == y) return;
    if (size[x] < size[y]) swap(x, y);
    pa[y] = x;
    size[x] += size[y];
}
```
